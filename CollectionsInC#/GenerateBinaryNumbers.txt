using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        int N = 5; // Number of binary numbers to generate
        List<string> binaryNumbers = GenerateBinaryNumbers(N);

        // Print result
        Console.WriteLine("Binary Numbers: " + string.Join(", ", binaryNumbers));
    }

    static List<string> GenerateBinaryNumbers(int N)
    {
        List<string> result = new List<string>(); // Store binary numbers
        Queue<string> queue = new Queue<string>();

        queue.Enqueue("1"); // Start with "1"

        for (int i = 0; i < N; i++)
        {
            string current = queue.Dequeue(); // Get front element
            result.Add(current); // Store in result list

            // Generate next binary numbers and enqueue them
            queue.Enqueue(current + "0");
            queue.Enqueue(current + "1");
        }

        return result;
    }
}
