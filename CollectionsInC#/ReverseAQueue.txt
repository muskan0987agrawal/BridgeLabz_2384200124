using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        // Define a queue
        Queue<int> queue = new Queue<int>();
        queue.Enqueue(10);
        queue.Enqueue(20);
        queue.Enqueue(30);

        // Reverse the queue
        Queue<int> reversedQueue = ReverseQueue(queue);

        // Print result
        Console.WriteLine("Reversed Queue: " + string.Join(", ", reversedQueue));
    }

    static Queue<int> ReverseQueue(Queue<int> queue)
    {
        Stack<int> stack = new Stack<int>();

        // Step 1: Dequeue all elements from queue and push into stack
        while (queue.Count > 0)
        {
            stack.Push(queue.Dequeue());
        }

        // Step 2: Pop all elements from stack and enqueue back into queue
        while (stack.Count > 0)
        {
            queue.Enqueue(stack.Pop());
        }

        return queue;
    }
}
